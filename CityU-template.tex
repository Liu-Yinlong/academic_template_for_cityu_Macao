\section{Drawing and Ploting}
\begin{frame}
	\frametitle{Ploting 2D Using PGF/TikZ}
\begin{tikzpicture}
	\begin{axis}
		[grid=major,
		height=6cm,
		width=8cm,]
		
		\addplot [
		blue,
		quiver={u=1,v=2*x},
		-stealth,
		samples=15,
		] {x^2};
	\end{axis}
\end{tikzpicture}
\end{frame}

\begin{frame}
	\frametitle{Ploting 3D Using PGF/TikZ}
	\begin{tikzpicture}
		\begin{axis}[
			grid=major,
			view={210}{30},
			]
			\addplot3+ [
			mesh,
			scatter,
			samples=10,
			domain=0:1,
			] {5*x*sin(2*deg(x)) * y*(1-y)};
		\end{axis}
	\end{tikzpicture}
